[docs]class ClientCommand(amp.Command):
    """Command for uiconnection.py->Daemon.command
     to remotely call guiplex.py->GUI.command.
     """
    arguments = [(b'handle', amp.Unicode()),
                 (b'command', amp.Unicode())]
